import os import json from openai import OpenAI from datetime import datetime # ========================================== # TechGuideChina - 全自动内容生成脚本 # 这个脚本会调用 AI 生成文章，并写入 index.html # ========================================== # 1. 配置 AI API (默认兼容 OpenAI 格式，可以使用各种便宜的中转API或DeepSeek) api_key = os.environ.get("AI_API_KEY") api_base = os.environ.get("AI_API_BASE", "https://api.moonshot.cn/v1") # 这里已经为你默认填好了 Kimi 的官方 API 地址 client = OpenAI( api_key=api_key, base_url=api_base ) # 2. 给 AI 的核心提示词 (让它写出高点击率的英文内容) prompt = """ You are an expert tech blogger and SEO specialist. Write a short, engaging snippet for a new blog post about a trending AI tool, AI automation strategy, or making money with AI. Output ONLY a valid JSON object with the following keys: "title": string, Catchy, high-CTR SEO title (e.g., "How to Automate..."). "category": string, 1-2 words max (e.g., "Make Money", "AI Tools", "SEO"). "description": string, 2 sentences max summarizing the content persuasively. "read_time": string, e.g., "3 min". """ try: # 3. 请求 AI print("正在呼叫 Kimi 生成内容...") response = client.chat.completions.create( model="moonshot-v1-8k", # 已经为你替换为 Kimi 的模型 messages=[ {"role": "system", "content": "You must output ONLY valid JSON."}, {"role": "user", "content": prompt} ], response_format={ "type": "json_object" }, # 强制要求输出 JSON temperature=0.7 ) content_str = response.choices[0].message.content data = json.loads(content_str) print(f"成功生成文章: {data['title']}") # 4. 将 AI 生成的内容组装成漂亮的 HTML 卡片 new_article_html = f"""
{data['category']} · Today

{data['title']}

{data['description']}

{data['read_time']} read
""" # 5. 读取 index.html，找到锚点并插入新文章 with open('index.html', 'r', encoding='utf-8') as f: html_content = f.read() # 寻找我们在 html 中留下的锚点替换位 anchor = "" if anchor in html_content: html_content = html_content.replace(anchor, f"{anchor}\n{new_article_html}") with open('index.html', 'w', encoding='utf-8') as f: f.write(html_content) print("成功将新文章写入 index.html！") else: print("错误：在 index.html 中找不到锚点 ") except Exception as e: print(f"发生错误: {e}")
